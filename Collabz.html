4. Smart Waste & Recycling Management

Schedules waste pickup based on urgency and fill-level. Uses BFS to check bin reachability and min-heap to prioritize pickups.

Contributors: Sujana + Khushi

Merged Ideas: Sujana: Smart waste management; Khushi: Waste-Heat Energy Recycling

Algorithms/Data Structures: Min-Heap + BFS

#include <bits/stdc++.h>
using namespace std;

struct Job {
    int urgency;
    int dist;
    int binId;
};
struct Cmp {
    bool operator()(const Job &a, const Job &b) const {
        if(a.urgency != b.urgency) return a.urgency < b.urgency;
        return a.dist > b.dist;
    }
};

int main(){
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    int n_bins; cout << "Enter number of bins: "; cin >> n_bins;
    struct Bin { int id; int fill; int urgency; int node; };
    vector<Bin> bins;
    cout << "For each bin: id fill_level(0-100) urgency(0-100) nodeIndex\n";
    for(int i=0;i<n_bins;i++){ Bin b; cin >> b.id >> b.fill >> b.urgency >> b.node; bins.push_back(b); }
    int nodes, edges; cout << "Enter number of street-nodes and edges: "; cin >> nodes >> edges;
    vector<vector<int>> g(nodes);
    cout << "Enter edges (u v) 0-indexed:\n";
    for(int i=0;i<edges;i++){ int u,v; cin >> u >> v; g[u].push_back(v); g[v].push_back(u); }
    int depotNode; cout << "Enter depot node index: "; cin >> depotNode;

    vector<int> dist(nodes,-1); queue<int> q; q.push(depotNode); dist[depotNode]=0;
    while(!q.empty()){ int u=q.front(); q.pop(); for(int v: g[u]) if(dist[v]==-1){ dist[v]=dist[u]+1; q.push(v); } }

    priority_queue<Job, vector<Job>, Cmp> pq;
    for(auto &b: bins){ int d = (b.node >=0 && b.node<nodes) ? dist[b.node] : -1;
        if(d==-1) cout << "Warning: bin " << b.id << " node " << b.node << " unreachable from depot\n";
        pq.push({b.urgency, (d==-1?1e9:d), b.id});
    }

    cout << "\nPickup schedule (most urgent first):\n";
    while(!pq.empty()){ auto j=pq.top(); pq.pop();
        cout << "Bin " << j.binId << " | urgency=" << j.urgency;
        if(j.dist >= 1e9) cout << " | unreachable\n"; else cout << " | hops_from_depot=" << j.dist << "\n";
    }
    return 0;
}

    int V,E; cout << "\nEnter number of machines/nodes for maintenance graph: "; cin >> V;
    cout << "Enter number of edges (u v weight) - weight can be negative: "; cin >> E;
    vector<Edge> edges(E); cout << "Enter edges (u v w):\n";
    for(int i=0;i<E;i++) cin >> edges[i].u >> edges[i].v >> edges[i].w;
    int source; cout << "Enter source node: "; cin >> source;
    vector<long long> dist; bool ok=bellmanFord(V,edges,source,dist);
    if(!ok) cout << "Negative cycle detected in maintenance graph\n";

    }
    return 0;
}
