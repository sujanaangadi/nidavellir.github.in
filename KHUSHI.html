2) AI-Based Worker Safety Monitoring

Algorithms Used: Dijkstra, BFS, Priority Queue

Explanation:
  • Factory floor modeled as a weighted graph where edges represent hazard levels.
  • Dijkstra computes safest evacuation route with minimum danger cost.
  • BFS models real-time hazard spread (fire, smoke, gas leaks).
  • Heap prioritizes critical danger zones for quicker alerts.
Time Complexity: O(E log V)
Show Code 
#include 
using namespace std;

vector dijkstra(int n,int src,vector>> &g){
    vector dist(n,INT_MAX);
    priority_queue,vector>,greater>> pq;
    dist[src]=0; 
    pq.push({0,src});

    while(!pq.empty()){
        auto [d,u]=pq.top(); pq.pop();
        if(d>dist[u]) continue;
        for(auto &p:g[u]){
            int v=p.first, w=p.second;
            if(dist[u]+w < dist[v]){
                dist[v]=dist[u]+w;
                pq.push({dist[v],v});
            }
        }
    }
    return dist;
}

3) Predictive Maintenance for Industrial Machines

Algorithms Used: Segment Tree, DFS, Heap

Explanation:
  • Segment tree stores vibration data and detects sudden abnormal spikes.
  • DFS identifies component dependency when a part begins failing.
  • Heap ranks machines by risk score for scheduled maintenance.
  • Helps prevent breakdown by early prediction of faults.
Time Complexity: O(log n)
Show Code 
#include 
using namespace std;

int build(vector &a, vector &st, int i,int l,int r){
    if(l==r) return st[i]=a[l];
    int m=(l+r)/2;
    return st[i]=max(build(a,st,2*i+1,l,m),
                     build(a,st,2*i+2,m+1,r));
}

int query(vector &st,int i,int l,int r,int ql,int qr){
    if(qrr) return -1;
    if(ql<=l && r<=qr) return st[i];
    int m=(l+r)/2;
    return max(query(st,2*i+1,l,m,ql,qr),
               query(st,2*i+2,m+1,r,ql,qr));
}

vector boxes = {100, 300, 500};

int findBox(int v){
    int low=0, high=boxes.size()-1, ans=-1;
    while(low<=high){
        int mid=(low+high)/2;
        if(boxes[mid] >= v){
            ans = mid;
            high = mid - 1;
        } else low = mid + 1;
    }
    return ans;
}
