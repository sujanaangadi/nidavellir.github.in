1) Smart Textile Quality Detector

Algorithms Used: KMP Pattern Matching, BFS/DFS

Explanation:
  • Industries scan fabric surfaces for repeated defect patterns such as holes or weaving errors.
  • KMP Pattern Matching helps detect patterns efficiently without rechecking characters.
  • BFS/DFS can map how defects are spreading across the cloth area.
  • This automation reduces manual inspection time and increases textile quality.
Time Complexity: O(n + m)
Show Code

vector buildLPS(vector& pat){
    vector lps(pat.size(), 0);

    auto lps = buildLPS(pat);
    int i=0, j=0;
    while(i < text.size()){
        if(text[i] == pat[j]){ i++; j++; }
        else if(j) j = lps[j-1];
        else i++;

}

2) AI-Based Worker Safety Monitoring

Algorithms Used: Dijkstra’s Algorithm, Priority Queue

Explanation:
  • Industrial areas are treated as interconnected nodes (factory map).
  • During accidents (gas leak/fire), workers must reach the safest exit.
  • Dijkstra’s algorithm finds the fastest and safest escape route.
  • Priority queue ensures the shortest paths are explored first.
Time Complexity: O(E log V)
Show Code 
#include 
using namespace std;

vector dijkstra(int n, int src, vector>>& adj){
    vector dist(n, INT_MAX);
    priority_queue, vector>, greater>> pq;

    dist[src] = 0;
    pq.push({0, src});

    while(!pq.empty()){
        auto [d,u] = pq.top(); pq.pop();
        if(d > dist[u]) continue;

        for(auto &e : adj[u]){
            int v=e.first, w=e.second;
            if(dist[u] + w < dist[v]){
                dist[v] = dist[u] + w;
                pq.push({dist[v], v});
            }
        }
    }
    return dist;
}

int main(){
    int n = 5;
    vector>> adj(n);
    adj[0]={{1,3},{2,1}};
    adj[1]={{3,4}};
    adj[2]={{3,2}};
    adj[3]={{4,1}};

    auto safe = dijkstra(n, 0, adj);
    for(int d : safe) cout << d << " ";
}

3) Predictive Maintenance for Industrial Machines

Algorithms Used: Segment Tree, Max Query

Explanation:
  • Industrial machines generate sensor readings (heat/vibration).
  • Segment Tree helps check vibration spikes in any time range.
  • This predicts breakdowns early and reduces maintenance cost.
  • Prevents machine downtime and improves safety.
Time Complexity: O(log n) per query
Show Code 
#include 
using namespace std;

int build(vector& a, vector& st, int i, int l, int r){
    if(l==r) return st[i]=a[l];
    int m=(l+r)/2;
    return st[i]=max(build(a,st,2*i+1,l,m),
                     build(a,st,2*i+2,m+1,r));
}

int query(vector& st, int i, int l, int r, int ql, int qr){
    if(qrr) return -1;
    if(ql<=l && r<=qr) return st[i];
    int m=(l+r)/2;
    return max(query(st,2*i+1,l,m,ql,qr),
               query(st,2*i+2,m+1,r,ql,qr));
}

int main(){
    int n; cin>>n;
    vector vib(n);
    for(int &x:vib) cin>>x;

    vector st(4*n);
    build(vib,st,0,0,n-1);

    int L,R; cin>>L>>R;
    int mx = query(st,0,0,n-1,L,R);

    cout << (mx>100 ? "Maintenance Needed" : "Normal");
}

6) Waste-Heat Energy Recycling System

Algorithms Used: Max Heap, Threshold Calculation

Explanation:
  • Industries generate waste heat from machines and boilers.
  • A max heap stores heat readings so hottest sources are processed first.
  • Heat above threshold (40°C) is converted to usable energy.
  • This reduces electricity usage and improves energy efficiency.
Time Complexity: O(n log n)
Show Code 
#include 
using namespace std;

int main(){
    int n;
    cin >> n;

    priority_queue pq;
    for(int i=0;i> t;
        pq.push(t);
    }

    float total = 0;
    while(!pq.empty()){
        float t = pq.top();
        pq.pop();

        float energy = max(0.0f, (t - 40) * 0.6f);
        cout << "Temp " << t << "°C -> " << energy << " kW\n";
        total += energy;
    }

    cout << "Total Recycled Energy: " << total << " kW";
}

7) AI-Based Product Packaging Optimization

Algorithms Used: Binary Search, Sorting

Explanation:
  • Products require specific box sizes based on volume.
  • Box sizes are sorted to enable binary search.
  • Binary search finds the smallest possible suitable box.
  • This reduces material waste and speeds packing operations.
Time Complexity: O(log n)
Show Code 
#include 
using namespace std;

vector boxes = {100, 300, 500};

int findBox(int v){
    int low=0, high=boxes.size()-1, ans=-1;
    while(low<=high){
        int mid=(low+high)/2;
        if(boxes[mid] >= v){
            ans = mid;
            high = mid - 1;
        } else low = mid + 1;
    }
    return ans;
}

int main(){
    int n;
    cin >> n;

    while(n--){
        int v;
        cin >> v;
        int idx = findBox(v);

        if(idx == -1) cout << "No suitable box\n";
        else cout << "Use box: " << boxes[idx] << " ml\n";
    }
}

9) Factory Pollution Monitoring & Control Dashboard

Algorithms Used: Segment Tree, Range Max Query

Explanation:
  • Pollution levels recorded from different factory zones.
  • Segment Tree allows fast detection of pollution spikes.
  • Helps identify dangerous areas that need urgent ventilation.
  • Improves worker safety and environmental compliance.
Time Complexity: O(log n)
Show Code 
#include 
using namespace std;

int build(vector& a, vector& st, int i, int l, int r){
    if(l==r) return st[i] = a[l];
    int m = (l+r)/2;
    return st[i] = max(
        build(a, st, 2*i+1, l, m),
        build(a, st, 2*i+2, m+1, r)
    );
}

int query(vector& st, int i, int l, int r, int L, int R){
    if(R < l || L > r) return -1;
    if(L <= l && r <= R) return st[i];
    int m = (l+r)/2;
    return max(
        query(st, 2*i+1, l, m, L, R),
        query(st, 2*i+2, m+1, r, L, R)
    );
}

int main(){
