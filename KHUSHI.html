1) Smart Textile Quality Detector

Purpose: Detects defects in fabric using sensor values.

Show C++ Code 
#include <iostream>
#include <vector>
using namespace std;

int main() {
    vector fabric = {100,102,98,150,101,99};
    int expected = 100, threshold = 20;

    for(int i=0;i<fabric.size();i++){
        if(abs(fabric[i]-expected) > threshold)
            cout<<"Defect at position " <<i<<"\n";
    }
}
