#include 
using namespace std;

vector dijkstra(int n, int src, vector>>& adj){
    vector dist(n, INT_MAX);
    priority_queue, vector>, greater>> pq;

    dist[src]=0;
    pq.push({0,src});

    while(!pq.empty()){
        auto [d,u] = pq.top(); pq.pop();
        if(d>dist[u]) continue;

        for(auto &p : adj[u]){
            int v = p.first, w = p.second;
            if(dist[u] + w < dist[v]){
                dist[v] = dist[u] + w;
                pq.push({dist[v], v});
            }
        }
    }
    return dist;
}
