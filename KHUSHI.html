2) AI-Based Worker Safety Monitoring

Algorithms Used:
  • Dijkstra’s Shortest Path
  • Binary Search (Safety Thresholds)
  • Heap-Based Priority Ranking
  • BFS/DFS Hazard Spread
How the Algorithm Works:
  1. Factory layout is treated as a weighted graph.
  2. Dijkstra finds the safest escape route from the worker's location.
  3. Heap prioritizes hazardous zones that need urgent alerts.
  4. BFS/DFS simulate spread of fire or gas leakage.
  5. Binary Search checks if safety level exceeds allowed threshold.
Efficiency Analysis:
Time Complexity: O((V + E) log V) (Dijkstra)
Space Complexity: O(V) Show Code 
#include 
using namespace std;

vector dijkstra(int n, int src, vector>> &adj){
    vector dist(n, INT_MAX);
    priority_queue, vector>, greater>> pq;

    dist[src] = 0;
    pq.push({0,src});

    while(!pq.empty()){
        auto [d,u] = pq.top(); pq.pop();
        if(d > dist[u]) continue;

        for(auto &e : adj[u]){
            int v=e.first, w=e.second;
            if(dist[u] + w < dist[v]){
                dist[v] = dist[u] + w;
                pq.push({dist[v], v});
            }
        }
    }
    return dist;
}

    int n = 5;
    vector>> adj(n);

    adj[1]={{3,4}};
    adj[2]={{3,2}};
    adj[3]={{4,1}};

    auto safe = dijkstra(n,0,adj);

    for(int d:safe) cout << d << " ";
}

3) Predictive Machine Maintenance

Algorithms Used:
  • Segment Tree (Range Max Query)
  • Heap (Risk Prioritization)
  • KMP (Repeated Fault Pattern Detection)
How the Algorithm Works:
  1. Machine vibration logs are stored in an array.
  2. A segment tree quickly finds maximum vibration in any time interval.
  3. If max vibration > threshold, maintenance alert is triggered.
  4. Heap ranks machines based on severity of vibration.
Efficiency Analysis:
Time Complexity: O(log n) (Segment Tree Query)
Space Complexity: O(n) Show Code 
#include 
using namespace std;

int build(vector& a, vector& st, int i, int l, int r){
    if(l==r) return st[i]=a[l];
    int m=(l+r)/2;
    return st[i]=max(build(a,st,2*i+1,l,m),
                     build(a,st,2*i+2,m+1,r));
}

int query(vector& st, int i, int l, int r, int ql, int qr){
    if(qrr) return -1;
    if(ql<=l && r<=qr) return st[i];
    int m=(l+r)/2;
    return max(query(st,2*i+1,l,m,ql,qr),
               query(st,2*i+2,m+1,r,ql,qr));
}

int main(){
    int n; cin>>n;
    vector vib(n);
    for(int &x:vib) cin>>x;

    vector st(4*n);
    build(vib,st,0,0,n-1);

    int L,R; cin>>L>>R;

    int mx=query(st,0,0,n-1,L,R);

    cout << (mx>100? "Maintenance Needed":"Normal");
}

7) AI-Based Packaging Optimization

Algorithms Used:
  • Binary Search
  • Quick Sort
  • Dynamic Programming
How the Algorithm Works:
  1. Box volumes stored in sorted array.
  2. Binary Search finds best-fit box.
  3. DP minimizes packaging cost.
Efficiency Analysis:
Time Complexity: O(log n)
Space Complexity: O(1) Show Code 
#include 
using namespace std;

vector boxes = {100, 300, 500};

int findBox(int v){
    int low=0, high=boxes.size()-1, ans=-1;
    while(low<=high){
        int mid=(low+high)/2;
        if(boxes[mid] >= v){
            ans=mid; high = mid - 1;
        }
        else low = mid + 1;
    }
    return ans;
}

int main(){
    int n; cin>>n;
    while(n--){
        int v; cin>>v;
        int idx=findBox(v);

        if(idx==-1) cout<<"No Box\n";
        else if(idx==0) cout<<"Small\n";
        else if(idx==1) cout<<"Medium\n";
        else cout<<"Large\n";
    }
}
