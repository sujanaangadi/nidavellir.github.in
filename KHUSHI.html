#include 
using namespace std;

float convertEnergy(float t){
    float base=40, f=0.6;
    float e=(t-base)*f;
    return (e<0)?0:e;
}

int main(){
    int n; cin >> n;
    float total=0;

    while(n--){
        float x; cin >> x;
        float e = convertEnergy(x);
        total += e;
        cout << "Converted: " << e << " kW\n";
    }

    cout << "Total Energy: " << total << " kW";
}
