1) Smart Traffic Signal System

Algorithms Used:
  • Max-Heap → Selects lane with highest priority
  • Greedy Decision → Emergency > High Traffic
  • Dynamic Priority Rebalancing
  • Lookup Table → Safety Minimum Wait
Efficiency Analysis:
Time Complexity: O(n)
Space Complexity: O(n)

#include 
using namespace std;

struct Lane {
    int lane_no;
    int vehicle_count;
    bool emergency;
};

struct Compare {
    bool operator()(Lane const &a, Lane const &b) {
        if (a.emergency != b.emergency)
            return b.emergency > a.emergency;
        return a.vehicle_count < b.vehicle_count;
    }
};

int main() {
    int n;
    cin >> n;

    priority_queue, Compare> pq;

    for (int i = 1; i <= n; i++) {
        int count;
        char emer;
        cin >> count >> emer;

        pq.push({i, count, emer=='y' || emer=='Y'});
    }

    Lane top = pq.top();
    cout << "Lane GREEN: " << top.lane_no;
}

  • Lookup Table → Direct waste → category mapping

#include 
#include 
using namespace std;

int classifyWaste(string w) {
    if (w=="plastic" || w=="polythene") return 1;
    if (w=="organic" || w=="banana peel") return 2;
    if (w=="paper" || w=="cardboard") return 3;
    if (w=="metal" || w=="aluminium") return 4;
    return 0;
}

int main() {
    unordered_map bins = {
        {1,"Plastic Bin"}, {2,"Organic Bin"},
        {3,"Paper Bin"}, {4,"Metal Bin"}
    };

    string w;
    getline(cin, w);
    int cat = classifyWaste(w);

    if (cat == 0) cout << "Manual Inspection";
    else cout << "Assigned Bin: " << bins[cat];
}
