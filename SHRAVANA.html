1) AI Smart Power Distribution Controller

Greedy Load Allocation
Algorithm used: Greedy — satisfy homes first, then industry with remaining energy.
How it works:

This greedy algorithm always gives priority to home demand first, then allocates leftover energy to industries.

Efficiency Analysis:

Time Complexity: O(1) — only arithmetic operations. Space Complexity: O(1).

Mapped SDG:

SDG 7 – Affordable & Clean Energy

C++ Code — Greedy Load Allocation
#include <bits/stdc++.h>
using namespace std;

int main(){
    double solar = 300.0, wind = 200.0, powerhouse = 250.0;
    double totalEnergy = solar + wind + powerhouse;

    double demandHomes = 500.0, demandIndustry = 400.0;

    cout << "Total energy: " << totalEnergy << "\n";

    double supplyHomes = min(totalEnergy, demandHomes);
    totalEnergy -= supplyHomes;

    double supplyIndustry = min(totalEnergy, demandIndustry);

    cout << "Supplied to homes: " << supplyHomes << "\n";
    cout << "Supplied to industry: " << supplyIndustry << "\n";
    cout << "Unserved demand: "
         << (demandHomes + demandIndustry - (supplyHomes + supplyIndustry)) << "\n";

    return 0;
}

2) Solar Panel Efficiency Analyzer

Sliding Window (Moving Average)
Detect drop in efficiency using moving average.
How it works:

Uses a sliding window to compute moving average of efficiency values to detect dust or shading issues.

Efficiency Analysis:

Time Complexity: O(n). Space Complexity: O(w) where w = window size.

Mapped SDG:

SDG 13 – Climate Action

C++ Code — Sliding Window
#include <bits/stdc++.h>
using namespace std;

int main(){
    vector dailyEfficiency = {95,94,93,78,76,75,74,92,93};
    int window = 3;
    double threshold = 80.0;

    deque dq;
    double sum = 0;

    for(size_t i=0;i window){
            sum -= dq.front();
            dq.pop_front();
        }

        if(dq.size() == window){
            double avg = sum / window;
            cout << "Day " << i+1 << " moving avg: " << avg << "%";
            if(avg < threshold) cout << "  <-- ALERT: Cleaning needed";
            cout << "\n";
        }
    }
    return 0;
}

3) Wind Turbine RPM Controller

Binary Search
How it works:

Binary search finds blade angle where RPM becomes equal to target RPM.

Efficiency Analysis:

Time Complexity: O(log n). Space Complexity: O(1).

Mapped SDG:

SDG 7 – Affordable & Clean Energy

C++ Code — Binary Search
#include <bits/stdc++.h>
using namespace std;

double rpm_at_angle(double angle, double wind_speed){
    return wind_speed * (100.0 * cos(angle * M_PI / 180.0)) - 0.5 * angle;
}

int main(){
    double wind = 8.0;
    double target = 450.0;

    double lo = 0, hi = 60;

    for(int i=0;i<40;i++){
        double mid = (lo + hi) / 2;
        double r = rpm_at_angle(mid, wind);

        if(r >= target) lo = mid;
        else hi = mid;
    }

    cout << "Angle: " << (lo+hi)/2 << "\n";
    cout << "RPM: " << rpm_at_angle((lo+hi)/2, wind) << "\n";
}

4) Battery Storage Optimization

0/1 Knapsack (DP)
How it works:

Dynamic Programming decides which batteries to charge to maximize storage benefit within capacity.

Efficiency Analysis:

Time Complexity: O(n * capacity). Space Complexity: O(capacity).

Mapped SDG:

SDG 12 – Responsible Consumption & Production

C++ Code — 0/1 Knapsack
#include <bits/stdc++.h>
using namespace std;

int main(){
    vector value = {10,40,30,50,35};
    vector cost  = {1,3,2,4,2};
    int cap = 6;

    vector dp(cap+1,0);

    for(int i=0;i=cost[i]; --w){
            dp[w] = max(dp[w], dp[w-cost[i]] + value[i]);
        }
    }

    cout << "Max benefit: " << dp[cap];
}

6) Energy Emergency Auto-Switching

Max-Heap
How it works:

Chooses largest available energy sources first using a max-heap.

Efficiency Analysis:

Time Complexity: O(n log n). Space Complexity: O(n).

Mapped SDG:

SDG 11 – Sustainable Cities & Communities

C++ Code — Max-Heap
#include <bits/stdc++.h>
using namespace std;

int main(){
    vector> b = {
        {50,"BatteryA"}, {120,"BatteryB"},
        {30,"Diesel1"}, {80,"BatteryC"}
    };

    priority_queue> pq;
    for(auto &x : b)
        pq.push(x);

    int req = 180;

    while(req > 0 && !pq.empty()){
        auto t = pq.top(); pq.pop();
        int give = min(req, t.first);

        cout << "Using " << t.second << " = " << give << " kWh\n";
        req -= give;
    }

    if(req > 0) cout << "Shortfall: " << req;
    else cout << "All covered";
}

8) Electricity Price Optimizer

Binary Search
How it works:

Binary search finds the maximum price such that demand does not exceed safe load.

Efficiency Analysis:

Time Complexity: O(log n). Space Complexity: O(1).

Mapped SDG:

SDG 8 – Decent Work & Economic Growth

C++ Code — Binary Search Price
#include <bits/stdc++.h>
using namespace std;

double demand(double p){
    return max(0.0, 1000.0 - 15.0 * p);
}

int main(){
    double safe = 700;
    double lo=0, hi=10;

    for(int i=0;i<50;i++){
        double mid = (lo+hi)/2;
        if(demand(mid) > safe)
            lo = mid;
        else
            hi = mid;
    }

    cout << "Price: " << (lo+hi)/2 << "\n";
    cout << "Demand: " << demand((lo+hi)/2);
}

9) House-to-House Energy Sharing

Greedy Matching (Improved Code)
How it works:

Houses with surplus energy distribute to houses with deficit. Surpluses sorted descending, deficits sorted descending. Matching done greedily.

Efficiency Analysis:

Time Complexity: O(n log n). Space Complexity: O(n).

Mapped SDG:

SDG 7 – Affordable & Clean Energy

C++ Code — Greedy Energy Sharing (Corrected)
#include <bits/stdc++.h>
using namespace std;

int main(){
    vector> houses = {
        {1,  50},   // positive = surplus
        {2, -30},   // negative = deficit
        {3, -40},
        {4,  20},
        {5, -10}
    };

    vector> surplus, deficit;

    for(auto &h : houses){
        if(h.second > 0) surplus.push_back(h);
        else deficit.push_back({h.first, -h.second});
    }

    sort(surplus.begin(), surplus.end(),
         [](auto &a, auto &b){ return a.second > b.second; });

    sort(deficit.begin(), deficit.end(),
         [](auto &a, auto &b){ return a.second > b.second; });

    int i = 0, j = 0;

    while(i < surplus.size() && j < deficit.size()){
        int give = min(surplus[i].second, deficit[j].second);

        cout << "Transfer " << give
             << " from House " << surplus[i].first
             << " to House "   << deficit[j].first << "\n";

        surplus[i].second -= give;
        deficit[j].second -= give;

        if(surplus[i].second == 0) i++;
        if(deficit[j].second == 0) j++;
    }

    return 0;
}

10) Smart Load Control

Priority Queue
How it works:

Devices with highest priority × power are turned off first until load goes below the limit.

Efficiency Analysis:

Time Complexity: O(n log n). Space Complexity: O(n).

Mapped SDG:

SDG 7 – Affordable & Clean Energy

C++ Code — Priority Queue

using namespace std;

struct Device {
    string name;
    int priority;

};

int main(){
    vector d = {
        {"Heater",8,2000},{"Fridge",1,150},{"Washer",7,1000},
        {"Lights",3,200},{"AC",9,2500},{"Computer",4,300}
    };

    int limit = 3000;
    int load = 0;

    for(auto &x : d) load += x.power;

    priority_queue> pq;

    for(int i=0;i off;

    while(load > limit && !pq.empty()){
        auto t = pq.top(); pq.pop();
        int id = t.second;

        load -= d[id].power;
        off.push_back(d[id].name);
    }

    cout << "Final load: " << load << "\n";
    for(string s : off)
        cout << "Turned off: " << s << "\n";
}
